skill_name: feature_api_architecture_pattern
version: 1.0

description: >
  Defines a modular feature-based .NET architecture using vertical slices,
  separating API, Application, Domain, Infrastructure, and EventBus layers.
  Supports Mediator-based internal events, CQRS-style queries, and external integration events.

---

core_principles:
  - Feature is a group of related use cases (NOT database tables)
  - Resource is a use-case boundary (NOT a single entity/table)
  - Commands mutate state; Queries read state
  - Domain events are internal and published via Mediator
  - Integration events are external and published via EventBus
  - API layer owns request/response mapping
  - Application layer owns business orchestration (NO DTO mapping)
  - Infrastructure owns persistence, cache, and external services

---

project_structure:

  Feature.Api.csproj:

    Apis:
      rule: "Entry point for HTTP/API layer per Resource"
      files:
        - "{Resource}sApi.cs"
        - "{Resource}sApiMappers.cs"
      responsibilities:
        - map Request → Command/Query/DomainEvent
        - map Entity → Response
        - define HTTP endpoints
        - use FeatureService for DI registration
      forbidden:
        - business logic
        - cache access
        - direct DB access

    Contracts:
      Requests:
        pattern: "*Request.cs"
      Responses:
        pattern: "*Response.cs"
      rule: "Pure transport models only"
      forbidden:
        - domain logic
        - infrastructure references

    Application:
      Commands:
        rule: "Write operations via Mediator"
        files:
          - "{Action}Command.cs"
          - "{Action}CommandHandler.cs"
        constraints:
          - must NOT use caching
          - may use IService or Context (DbContext)
          - must NOT map DTOs
        responsibilities:
          - create/update domain entities
          - publish DomainEvents via Mediator

      Queries:
        rule: "Read operations (CQRS)"
        files:
          - "I{Resource}Queries.cs"
          - "{Resource}Queries.cs"
          - "{Resource}QueryModels.cs"
        constraints:
          - may use ICacheService
          - TTL defined inside query layer
        responsibilities:
          - optimize read models
          - handle cache-aside pattern

      IntegrationEventHandlers:
        rule: "Handles external system events from EventBus"
        constraints:
          - no domain mutation without explicit command
          - may trigger commands

      DomainEventHandlers:
        rule: "Handles internal Mediator events"
        responsibilities:
          - cache invalidation
          - internal projections
          - side effects within service

      Mappers:
        rule: "Entity → Response ONLY"
        files:
          - "{Resource}ResponseMappers.cs"
        constraints:
          - must NOT map Request → Command
          - must NOT map Command → Entity

    Infrastructure:

      Persistence:
        file: "FeatureContext.cs"
        rule: "EF Core DbContext only"

      Services:
        ExternalService:
          structure:
            - I{External}Service.cs
            - {External}Service.cs
            - {External}ServiceModels.cs
            - {External}ServiceMapper.cs
          rule: "External system integration only"
          constraints:
            - isolated per external system

      Cache:
        files:
          - "ICacheService.cs"
          - "RedisCacheService.cs"
          - "CacheKeys.cs"
          - "CachePolicies.cs"
        key_format: "{Feature}:{Resource}:{Identifier|ParamsHash}"
        ttl_policies:
          short: "2 minutes"
          medium: "10 minutes"
          long: "6 hours"
        rules:
          - cache ONLY in Queries
          - invalidation ONLY in DomainEventHandlers

    Domain:

      Entity:
        rule: "Persistence models (DB objects)"

      Constants:
        rule: "Enums and shared constants only"

      DomainEvents:
        rule: "Internal events published via Mediator"

    Extensions:
      rule: "DI registration, utilities only"

    GlobalUsing.cs:
      rule: "shared imports only"

---

EventBus.csproj:

  Abstractions:
    - IEventBus.cs
    - IEventBusBuilder.cs
    - IIntegrationEventHandler.cs
    - IntegrationEvent.cs
    - EventBusSubscriptionInfo.cs

  Events:
    Feature:
      rule: "Integration events shared between services"
      naming: "{Actioned}IntegrationEvent.cs"
      constraints:
        - must be immutable contracts
        - versioning allowed only via new events
        - no domain logic inside events

  Extensions:
    rule: "utility helpers only"

---

event_model_rules:

  domain_event:
    definition: "Internal event inside a single service"
    publisher: "Mediator"
    consumers: "DomainEventHandlers"
    allowed_actions:
      - cache invalidation
      - internal workflows
    forbidden:
      - external communication

  integration_event:
    definition: "Event shared across microservices"
    publisher: "EventBus"
    consumers: "other services"
    allowed_actions:
      - cross-service communication
      - async workflows
    constraints:
      - must be version-safe
      - immutable contract

---

cache_rules:

  pattern: "cache-aside"

  flow:
    - Query checks cache first
    - On miss → DB fetch
    - Store result in cache
    - Return response

  invalidation:
    trigger: "DomainEvent via Mediator"
    rule: "NEVER invalidate cache in Commands"

---

api_rules:

  responsibilities:
    - HTTP routing
    - request validation
    - request → command/query mapping
    - entity → response mapping

  forbidden:
    - business logic
    - persistence logic
    - cache logic

---

application_rules:

  commands:
    - mutate state
    - publish domain events
    - no caching
    - no DTO mapping

  queries:
    - read optimization layer
    - cache allowed
    - return read models only

---

mapping_rules:

  api_layer:
    allowed:
      - Request → Command/Query
      - Entity → Response

  application_layer:
    forbidden:
      - DTO mapping
      - HTTP model awareness

  infrastructure_layer:
    allowed:
      - external model mapping
      - persistence mapping

---

output_behavior:

  when_generating_code:
    - enforce strict layer boundaries
    - never mix API and domain logic
    - never place caching outside infrastructure/application rules
    - prefer explicit over implicit dependencies
